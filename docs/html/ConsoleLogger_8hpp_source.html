107 void log(nixoncpp::logging::Level level, const std::string &message, const std::string &caller) {
108 std::lock_guard<std::mutex> lock(logMutex_);
109
110 // Get current time
111 auto now = std::chrono::system_clock::now();
112 std::time_t now_c = std::chrono::system_clock::to_time_t(now);
113 std::tm now_tm;
114
115 // Windows and POSIX have different thread-safe localtime functions
116#ifdef _WIN32
117 localtime_s(&now_tm, &now_c);
118#else
119 localtime_r(&now_c, &now_tm);
120#endif
121
122 // Create log header
123 std::ostringstream header;
124 if (!appPrefix_.empty()) {
125 header << "[" << appPrefix_ << "]";
126 }
127 header << "[" << std::put_time(&now_tm, "%Y-%m-%d %H:%M:%S") << "]";
128 header << "[" << levelToString(level) << "]";
129 if (!caller.empty()) {
130 header << "[" << caller << "]";
131 }
132 header << " ";
133
134 // Log to console
135 if (shouldUseColors()) {
136 setConsoleColor(level);
137 }
138 std::cout << header.str() << message;
139 if (addNewLine_) {
140 std::cout << "\n";
141 }
142
143 if (shouldUseColors()) {
144 resetConsoleColor();
145 }
146
147 // Log to file if enabled
148 if (logFile_.is_open()) {
149 logFile_ << header.str() << message;
150 if (addNewLine_) {
151 logFile_ << "\n";
152 }
153 logFile_.flush(); // Force immediate write to disk
154 }
155 };
